001package de.sgd.josm.plugins.osm2x.helper;
002
003import java.util.Vector;
004import java.util.concurrent.ThreadLocalRandom;
005
006import org.openstreetmap.josm.data.osm.DataSet;
007import org.openstreetmap.josm.data.osm.Node;
008
009public class IdGenerator {
010        private Vector<Long> ids;
011
012        public IdGenerator(DataSet ds) {
013                if (ds != null)
014                {
015                        ids = new Vector<>();
016                        for (Node n : ds.getNodes()) {
017                                ids.add(n.getId());
018                        }
019                }
020        }
021
022        public Long generateID() {
023                Long id = ThreadLocalRandom.current().nextLong(1000000, 1000000000);
024
025                while (ids.contains(id)) {
026                        id = ThreadLocalRandom.current().nextLong(1000000, 1000000000);
027                }
028
029                ids.add(id);
030                return id;
031        }
032
033
034}
