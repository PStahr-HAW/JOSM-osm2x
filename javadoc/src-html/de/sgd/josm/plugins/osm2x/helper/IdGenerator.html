001package de.sgd.josm.plugins.osm2x.helper;
002
003import java.util.ArrayList;
004import java.util.concurrent.ThreadLocalRandom;
005
006import org.openstreetmap.josm.data.osm.DataSet;
007import org.openstreetmap.josm.data.osm.Node;
008
009/**
010 * Generator for unique ids
011 */
012public class IdGenerator {
013        private ArrayList<Long> ids;    // vector containing all ids
014
015        /**
016         * Initialize new id generator
017         * @param ds DataSet to import ids from
018         */
019        public IdGenerator(DataSet ds) {
020                if (ds != null)
021                {
022                        // initialize new vector and add all ids from dataset
023                        ids = new ArrayList<>();
024                        for (Node n : ds.getNodes()) {
025                                ids.add(n.getId());
026                        }
027                }
028        }
029
030        /**
031         * Generate new unique id
032         * @return unique id
033         */
034        public Long generateID() {
035                // generate new random id in range from 1000000 to 1000000000
036                Long id = ThreadLocalRandom.current().nextLong(1000000, 1000000000);
037
038                while (ids.contains(id)) {
039                        // if id is already contained in dataset generate new
040                        id = ThreadLocalRandom.current().nextLong(1000000, 1000000000);
041                }
042
043                ids.add(id);    // add generated id to vector and return id
044                return id;
045        }
046}
