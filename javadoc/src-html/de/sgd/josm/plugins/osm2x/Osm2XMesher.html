033        public Osm2XMesher(DataSet ds) {
034                this.original_ds = ds;
035                this.str_to_double = Pattern.compile("(\\d+\\.?\\d*)");
036
037                // Check if all paths have the width attribute
038                boolean width_present = true;
039                for (Way w : ds.getWays()) {
040                        if (!w.hasTag("width")) {
041                                w.setHighlighted(true);
042                                width_present = false;
043                        }
044                }
045
046                if (!width_present) {
047                        this.ds = ds;
048                }
049        }

055        public boolean isDatasetComplete() {
056                // Check if all paths have the width attribute
057                boolean width_present = true;
058                for (Way w : ds.getWays()) {
059                        if (!w.hasTag("width") && !w.isDeleted()) {
060                                width_present = false;
061                        }
062                }
063                return width_present;
064        }

066        public void interpolate() {
067                DataSet ds = new DataSet(this.original_ds);
068                this.idGen = new IdGenerator(ds);
069
070                System.out.println("Start interpolation");
071
072                ds.beginUpdate();
073                try
074                {
075                        if (ds != null) {
076                                for (Way w : ds.getWays()) {
077
078                                        for (int i = 1; i < w.getRealNodesCount(); i++) {
079                                                Node last_node = w.getNode(i-1);
080                                                Node curr_node = w.getNode(i);
081
082                                                double dist = last_node.getCoor().greatCircleDistance(curr_node.getCoor());
083
084                                                if (dist > 5.0) {       // TODO threshold from properties
085                                                        int add_nodes = (int) Math.ceil(dist / 5);
086
087                                                        int k = 0;
088                                                        for (k = 1; k < add_nodes; k++) {
089                                                                // Calculate new latlon and node
090                                                                LatLon ll = last_node.getCoor().interpolate(curr_node.getCoor(), (double)k/add_nodes);
091                                                                Node nn = new Node(ll);
092                                                                nn.setOsmId(this.idGen.generateID(), 1);
093                                                                // add new node to dataset and way
094                                                                ds.addPrimitive(nn);
095                                                                w.addNode(i+k-1, nn);
096                                                        }
097                                                        i += k-1;
098                                                }
099                                        }
100                                }
101                        }
102                } catch (Exception e) {
103                        System.out.println(e.getMessage());
104                } finally
105                {
106                        ds.endUpdate();
107                }
108                this.interpolated_ds = ds;
109        }

111        public void createRoadNetwork() {
112                // get all ways -> ignore areas, but with warning
113                // ignore ways without width attribute
114                // ignore ways with attribute mesh=false
115                DataSet ds = (this.interpolated_ds == null) ? new DataSet(original_ds) : new DataSet(interpolated_ds);
116                this.idGen = new IdGenerator(ds);
117
118                System.out.println("Start meshing");
119
120                ds.beginUpdate();
121                try
122                {
123                        HashMap<Node, Osm2XNodeList> addedNodes = new HashMap<>();
124                        List<Way> addedWays = new ArrayList<>();
125
126                        for (Node n : ds.getNodes()) {
127                                Osm2XNodeList nodes = new Osm2XNodeList();
128
129                                if (n.isConnectionNode()) {
130                                        // node connects two or more ways
131                                        TreeMap<Double, Way> angles = getNeighbourNodes(n);
132                                        Entry<Double, Way> last_entry = null;
133                                        for (Entry<Double, Way> entry : angles.entrySet()) {
134                                                if (last_entry == null) {
135                                                        last_entry = entry;
136                                                        continue;
137                                                }
138
139                                                Node new_node = new Node(calculateNode(n.getCoor(), last_entry, entry));
140                                                new_node.setOsmId(this.idGen.generateID(), 1);
141                                                nodes.add(new_node, n.getCoor().bearing(new_node.getCoor()));
142                                                last_entry = entry;
143                                        }
144
145                                        Node new_node = new Node(calculateNode(n.getCoor(), last_entry, angles.firstEntry()));
146                                        new_node.setOsmId(this.idGen.generateID(), 1);
147                                        nodes.add(new_node, n.getCoor().bearing(new_node.getCoor()));
148                                        addedNodes.put(n, nodes);
149                                }
150                        }
151
152                        // compute parallel ways
153                        List<Node> more_nodes = new ArrayList<>();
154                        List<Way> more_ways = new ArrayList<>();
155                        for (Way w : ds.getWays()) {
156                                if (w.getRealNodesCount() < 1) continue;
157
158                                double width = 0;
159                                if (w.get("width") != null) {
160                                        width = parseDouble(w.get("width"));
161                                } else {
162                                        continue;
163                                }
164                                // TODO filtern??
165                                Way way_l = new Way(0);
166                                way_l.setKeys(w.getKeys());
167                                Map<String, String> keys = way_l.getKeys();
168                                keys.put("left", "true");
169                                way_l.setKeys(keys);
170                                // add outer node
171
172                                Way con_l1 = new Way(0);
173                                con_l1.setKeys(way_l.getKeys());
174                                con_l1.addNode(w.getNode(0));
175
176                                Way con_l2 = new Way(0);
177                                con_l2.setKeys(way_l.getKeys());
178                                // add outer node
179
180                                // ways for rights side
181                                Way way_r = new Way(0);
182                                way_r.setKeys(w.getKeys());
183                                // add outer node
184
185                                Way con_r1 = new Way(0);
186                                con_r1.setKeys(w.getKeys());
187                                con_r1.addNode(w.getNode(0));
188
189                                Way con_r2 = new Way(0);
190                                con_r2.setKeys(w.getKeys());
191                                // add outer node
192
193                                Way[] con_r = {con_r1, con_r2};
194                                Way[] con_l = {con_l1, con_l2};
195
196                                if (addedNodes.containsKey(w.getNode(0)) && w.getNodesCount() > 1) {
197                                        double ang1 = w.getNode(0).getCoor().bearing(w.getNode(1).getCoor());
198                                        double ang2 = ang1 > Math.PI ? ang1 - Math.PI : ang1 + Math.PI;
199
200                                        List<Node> r_nodes = addedNodes.get(w.getNode(0)).getRightNodesWithinAngle(ang1, ang2);
201                                        for (Node n : r_nodes) {
202                                                way_r.addNode(n);
203                                        }
204                                        if (r_nodes.size() > 0) {
205                                                con_r[1].addNode(r_nodes.get(r_nodes.size()-1));
206                                        }
207
208                                        List<Node> l_nodes = addedNodes.get(w.getNode(0)).getLeftNodesWithinAngle(ang1, ang2);
209                                        for (Node n : l_nodes) {
210                                                way_l.addNode(n);
211                                        }
212                                        if (l_nodes.size() > 0) {
213                                                con_l[1].addNode(l_nodes.get(l_nodes.size()-1));
214                                        }
215                                }
216
217                                for (int i = 1; i < w.getNodesCount()-1; i++) {
218                                        Node n_i = w.getNode(i);
219                                        double ang1 = n_i.getCoor().bearing(w.getNode(i+1).getCoor());
220                                        double ang2 = n_i.getCoor().bearing(w.getNode(i-1).getCoor());
221
222                                        if (addedNodes.containsKey(n_i)) {
223                                                // add nodes to way
224                                                for (Node n : addedNodes.get(n_i).getRightNodesWithinAngle(ang1, ang2)) {
225                                                        way_r.addNode(n);
226                                                        con_r[(i+1) % 2].addNode(n);
227                                                }
228
229                                                for (Node n : addedNodes.get(n_i).getLeftNodesWithinAngle(ang1, ang2)) {
230                                                        way_l.addNode(n);
231                                                        con_l[(i+1) % 2].addNode(n);
232                                                }
233
234                                                con_r[i % 2].addNode(n_i);
235                                                con_l[i % 2].addNode(n_i);
236                                        } else {
237                                                double ang = (ang1 > ang2) ? (ang1 + ang2) / 2 + Math.PI : (ang1 + ang2) / 2;
238
239                                                // Way on the right side
240                                                Node new_node = new Node(Osm2XConversions.calculatePositionFrom(n_i.getCoor(), ang, width/3));
241                                                new_node.setOsmId(this.idGen.generateID(), 1);
242                                                more_nodes.add(new_node);
243                                                way_r.addNode(new_node);
244
245                                                // Way on the left side
246                                                Node new_node1 = new Node(Osm2XConversions.calculatePositionFrom(n_i.getCoor(), ang + Math.PI, width/3));
247                                                new_node1.setOsmId(this.idGen.generateID(), 1);
248                                                more_nodes.add(new_node1);
249                                                way_l.addNode(new_node1);
250
251                                                con_r[(i+1) % 2].addNode(new_node);
252                                                con_r[i % 2].addNode(n_i);
253                                                con_l[(i+1) % 2].addNode(new_node1);
254                                                con_l[i % 2].addNode(n_i);
255                                        }
256                                }
257
258                                Node last_node = w.getNode(w.getNodesCount()-1);
259                                con_r[(w.getNodesCount()-1) % 2].addNode(last_node);
260                                con_l[(w.getNodesCount()-1) % 2].addNode(last_node);
261
262                                if (addedNodes.containsKey(last_node)) {
263                                        double ang2 = last_node.getCoor().bearing(w.getNode(w.getNodesCount()-2).getCoor());
264                                        double ang1 = ang2 > Math.PI ? ang2 - Math.PI : ang2 + Math.PI;
265
266                                        List<Node> r_nodes = addedNodes.get(last_node).getRightNodesWithinAngle(ang1, ang2);
267                                        for (Node n : r_nodes) {
268                                                way_r.addNode(n);
269                                        }
270                                        if (r_nodes.size() > 0) {
271                                                con_r[w.getNodesCount() % 2].addNode(r_nodes.get(0));
272                                        }
273
274                                        List<Node> l_nodes = addedNodes.get(last_node).getLeftNodesWithinAngle(ang1, ang2);
275                                        for (Node n : l_nodes) {
276                                                way_l.addNode(n);
277                                        }
278                                        if (l_nodes.size() > 0) {
279                                                con_l[w.getNodesCount() % 2].addNode(l_nodes.get(0));
280                                        }
281                                }
282
283                                more_ways.add(way_l);
284                                more_ways.add(con_l[0]);
285                                more_ways.add(con_l[1]);
286                                more_ways.add(way_r);
287                                more_ways.add(con_r[0]);
288                                more_ways.add(con_r[1]);
289                        }
290
291                        System.out.println("Add all nodes to dataset");
292                        for (Osm2XNodeList nodes : addedNodes.values()) {
293                                for (Node n : nodes.getNodelist()) {
294                                        ds.addPrimitive(n);
295                                }
296                        }
297                        System.out.println("Add all ways to dataset");
298                        for (Way w : addedWays) {
299                                ds.addPrimitive(w);
300                        }
301
302                        System.out.println("Add more nodes to dataset");
303                        for (Node n : more_nodes) {
304                                ds.addPrimitive(n);
305                        }
306                        System.out.println("Add more ways to dataset");
307                        for (Way w : more_ways) {
308                                ds.addPrimitive(w);
309                        }
310
311                } catch (Exception e) {
312                        System.out.println(e.getMessage());
313                } finally
314                {
315                        ds.endUpdate();
316                }
317                this.ds = ds;
318        }

324        private TreeMap<Double, Way> getNeighbourNodes(Node node) {
325                TreeMap<Double, Way> neighbours = new TreeMap<>();
326
327                for (Way w : node.getParentWays()) {
328                        for (Node n : w.getNeighbours(node)) {
329                                neighbours.put(node.getCoor().bearing(n.getCoor()), w);
330                        }
331                }
332                return neighbours;
333        }

335        /**
336         * Calculate a new node based on the width of way 1 and way 2.
337         * @param origin the node connecting both ways
338         * @param way1 the first way
339         * @param way2 the second way
340         * @return coordinates for the calculated node
341         */
342        private LatLon calculateNode(LatLon origin, Entry<Double, Way> way1, Entry<Double, Way> way2) {
343                // calculate distance in meters
344                double b1 = parseDouble(way1.getValue().get("width"));
345                double b2 = parseDouble(way2.getValue().get("width"));
346                double ang_diff = way2.getKey() - way1.getKey();
347                //System.out.print("Alpha: " + ang_diff);
348
349                double x = (b1 * Math.sin(way2.getKey()) + b2 * Math.sin(way1.getKey()))
350                                / 3.0 / Math.sin(ang_diff);
351
352                double y = (b1 * Math.cos(way2.getKey()) + b2 * Math.cos(way1.getKey()))
353                                / 3.0 / Math.sin(ang_diff);
354
355                //System.out.printf("xy = (%.4f, %.4f), ", x, y);
356                x = origin.lon() + x*Osm2XConversions.METER_TO_LATLON/Math.cos(origin.lat()/180*Math.PI);
357                y = origin.lat() + y*Osm2XConversions.METER_TO_LATLON;
358
359                //System.out.printf("lat/lon: %.7f, %.7f - ", y,x);
360
361                return new LatLon(y,x);
362        }

364        /**
365         *
366         * @param s
367         * @return
368         */
369        private double parseDouble(String s) {
370                double d = 0;
371                if (s == null) return d;
372                try {
373                        d = Double.valueOf(s);
374                } catch (NumberFormatException nfe) {
375                        // try to parse with regex
376                        Matcher matcher = str_to_double.matcher(s);
377                        //System.out.println("Try to match " + s);
378                        if (matcher.find()) {
379                                try {
380                                        String str = matcher.group();
381                                        //System.out.println("Found match " + str);
382                                        d = Double.valueOf(str);
383                                        //System.out.println("Double creation was successful: " + d);
384                                } catch (NumberFormatException nfe1) {}
385                        }
386                }
387                return d;
388        }
