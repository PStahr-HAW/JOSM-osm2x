162        /**
163         * Count nodes without 'highway', 'surface' and 'width' attribute
164         * @return number of nodes as string
165         */
166        protected String getNodeCount() {
167                DataSet ds;
168                if ((ds = MainApplication.getLayerManager().getEditDataSet()) != null) {
169                        int counter = 0;
170                        for (Way w : ds.getWays())
171                        {
172                                if (!w.isDeleted() && (!w.hasTag("highway") || !w.hasTag("surface")
173                                                || !w.hasTag("width")))
174                                {
175                                        counter++;
176                                }
177                        }
178
179                        return Integer.toString(counter);
180                }
181                return tr("-1");
182        }
183
184        /**
185         * Count nodes with tag 'entrance' and 'addr:street'
186         * @return
187         */
188        protected String getAddrCount() {
189                DataSet ds;
190                if ((ds = MainApplication.getLayerManager().getEditDataSet()) != null) {
191                        Collection<Node> nodes = ds.getNodes();
192
193                        int numNodes = 0;
194                        for (Node n : nodes)
195                        {
196                                if (n.hasTag("entrance") && n.hasTag("addr:street") && !n.isDeleted())
197                                {
198                                        numNodes++;
199                                }
200                        }
201
202                        return Integer.toString(numNodes);
203                }
204                return tr("-1");
205        }
206
207        /**
208         *
209         * @return
210         */
211        protected String getMissingAddrLinks() {
212                DataSet ds;
213                if ((ds = MainApplication.getLayerManager().getEditDataSet()) != null) {
214                        Collection<Node> nodes = ds.getNodes();
215
216                        int numNodes = 0;
217                        for (Node n : nodes)
218                        {
219                                if (n.hasTag("entrance") && !n.isDeleted()) {
220                                        boolean hasWay = false;
221                                        for (Way w : n.getParentWays()) {
222                                                if (w.hasTag("highway")) {
223                                                        hasWay = true;
224                                                        break;
225                                                }
226                                        }
227                                        numNodes = hasWay ? numNodes : numNodes+1;
228                                }
229                        }
230
231                        return Integer.toString(numNodes);
232                }
233                return tr("-1");
234        }

245        /**
246         * Create barriers and highways layer
247         */
248        protected void splitLayer() {
249                // check if layer highways and barriers is present
250                int answer = -1;
251                List<Layer> layers = MainApplication.getLayerManager().getLayers();
252                for (Layer l : layers)
253                {
254                        if (l.getName().equalsIgnoreCase("barriers") || l.getName().equalsIgnoreCase("highways"))
255                        {
256                                if (answer < 0)
257                                {
258                                        answer = JOptionPane.showConfirmDialog(MainApplication.getMainFrame(),
259                                                        "Layer with name 'barriers' or 'highways' detected. Do you want to continue and delete these layers?",
260                                                        "Confirm to delete layer", JOptionPane.YES_NO_OPTION);
261                                }
262                                if (answer == JOptionPane.YES_OPTION)
263                                {
264                                        MainApplication.getLayerManager().removeLayer(l);
265                                }
266                                else
267                                {
268                                        return;
269                                }
270                        }
271                }
272
273                // get current active dataset
274                DataSet ds = MainApplication.getLayerManager().getEditDataSet();
275                if (ds != null)
276                {
277                        // create ds_new as a copy from ds
278                        DataSet ds_new = new DataSet(ds);
279                        ds_new.beginUpdate();
280                        try {
281                                // remove all relations
282                                for (Relation r : ds_new.getRelations()) {
283                                        ds_new.removePrimitive(r.getPrimitiveId());
284                                }
285
286                                // copy address tags from building to entrance node
287                                if (Config.getPref().getBoolean(Osm2XConstants.PREF_COPY_ADDRESS, Osm2XConstants.DEFAULT_COPY_ADDRESS))
288                                {
289                                        for (Node n : ds_new.getNodes())
290                                        {
291                                                if (n.hasTag("entrance") && !n.hasTag("addr:street"))
292                                                {
293                                                        List<Way> refs = n.getParentWays();
294                                                        for (Way p : refs) {
295                                                                if (p.hasTag("addr:street") && p.hasTag("building")) {
296                                                                        n.put("addr:street", p.get("addr:street"));
297                                                                        n.put("addr:housenumber", p.get("addr:housenumber"));
298                                                                        n.put("addr:city", p.get("addr:city"));
299                                                                        n.put("addr:postcode", p.get("addr:postcode"));
300                                                                        break;
301                                                                }
302                                                        }
303                                                }
304                                        }
305                                }
306
307                                // create filter from preferences
308                                Filter f = new Filter(SearchSetting.readFromString("C " + Config.getPref().get(Osm2XConstants.PREF_HIGHWAY_FILTER, Osm2XConstants.DEFAULT_HIGHWAY_FILTER)));
309                                f.inverted = true;                                              // set inverted to true to get matching primitives
310                                FilterMatcher matcher = FilterMatcher.of(f);
311
312                                // remove all ways that do not match the specified tags
313                                Osm2XFilter.executeMatching(ds_new, matcher, Way.class::isInstance);
314                                // remove all nodes that do not match the filter
315                                Osm2XFilter.executeMatching(ds_new, matcher, Node.class::isInstance);
316
317                                matcher.reset();
318                        } catch (SearchParseError e) {
319                                // display warning
320                                JOptionPane.showMessageDialog(MainApplication.getMainFrame(), "SearchParseError: Check highway filtering and try again.", "SearchParseError", JOptionPane.WARNING_MESSAGE);
321                                e.printStackTrace();
322                                return;
323                        } finally {
324                                ds_new.endUpdate();
325                        }
326
327                        Layer highways = new OsmDataLayer(ds_new, "highways", null);
328                        MainApplication.getLayerManager().addLayer(highways);
329
330                        // create barrier layer
331                        // copy current dataset to ds_new
332                        ds_new = new DataSet(ds);
333                        ds_new.beginUpdate();
334                        try {
335                                // remove all relations
336                                for (Relation r : ds_new.getRelations()) {
337                                        ds_new.removePrimitive(r.getPrimitiveId());
338                                }
339
340                                // create filter
341                                Filter f = new Filter(SearchSetting.readFromString("C " + Config.getPref().get(Osm2XConstants.PREF_BARRIER_FILTER, Osm2XConstants.DEFAULT_BARRIER_FILTER)));
342                                f.inverted = true;
343                                FilterMatcher matcher = FilterMatcher.of(f);
344
345                                // remove all ways that do not match the specified tags
346                                Osm2XFilter.executeMatching(ds_new, matcher, Way.class::isInstance);
347                                // remove all nodes that do not match the filter
348                                Osm2XFilter.executeMatching(ds_new, matcher, Node.class::isInstance);
349
350                                matcher.reset();
351                        } catch (SearchParseError e) {
352                                JOptionPane.showMessageDialog(MainApplication.getMainFrame(), "SearchParseError: Check barrier filtering and try again.", "SearchParseError", JOptionPane.WARNING_MESSAGE);
353                                e.printStackTrace();
354                                return;
355                        } finally {
356                                ds_new.endUpdate();
357                        }
358                        Layer barrier = new OsmDataLayer(ds_new, "barriers", null);
359                        MainApplication.getLayerManager().addLayer(barrier);
360                }
361        }
