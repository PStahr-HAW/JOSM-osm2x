001package de.sgd.josm.plugins.osm2x.io.svg;
002
003public abstract class SvgShape {
004
005        String fill = "none";
006        String stroke = "none";
007        double stroke_width = 0.0;
008
009        abstract String toSvg(SvgPoint documentOrigin);
010
011        abstract double minLat();
012        abstract double maxLat();
013        abstract double minLon();
014        abstract double maxLon();
015
016        /**
017         * Set the fill color of the shape.
018         * @param color
019         */
020        void setFill(String color) {
021                this.fill = color;
022        }
023
024        /**
025         * Set stroke with specified color and width.
026         * @param color the color as string
027         * @param width the width in mm
028         */
029        void setStroke(String color, double width) {
030                this.stroke = color;
031                this.stroke_width = width;
032        }
033
034}
