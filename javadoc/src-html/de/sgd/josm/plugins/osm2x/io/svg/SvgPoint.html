001package de.sgd.josm.plugins.osm2x.io.svg;
002
003import de.sgd.josm.plugins.osm2x.helper.Osm2XConversions;
004
005public class SvgPoint {
006
007        private double lat;
008        private double lon;
009
010        public SvgPoint(double lat, double lon) {
011                this.lat = lat;
012                this.lon = lon;
013        }
014
015        public double lat() {
016                return this.lat;
017        }
018
019        public double lon() {
020                return this.lon;
021        }
022
023        /**
024         * Determines if the other point has almost the same lat/lon values.
025         * @param other the other point
026         * @param eps the acceptable difference in meters
027         * @return true if the difference is smaller than the acceptable difference otherwise false
028         */
029        public boolean equalsEps(SvgPoint other, double eps) {
030                double[] xy = this.getLocalCoordinates(other);
031
032                System.out.printf("Distance from %.7f, %.7f to %.7f, %.7f is %.6f, %.6f [m]\n",
033                                lat, lon, other.lat, other.lon, xy[0], xy[1]);
034
035                return Math.abs(xy[0]) < eps && Math.abs(xy[1]) < eps;
036        }
037
038        /**
039         * Calculates the coordinates of the point starting from the defined origin.
040         * The positive x-axis points to the right, the positive y-axis points down.
041         * @param origin
042         * @return an array containing the distances in meters
043         */
044        public double[] getLocalCoordinates(SvgPoint origin) {
045                double[] xy = {0.0,0.0};
046
047                xy[0] = (lon - origin.lon) / Osm2XConversions.METER_TO_LATLON * Math.cos((origin.lat + lat) * Math.PI/360);
048                xy[1] = (origin.lat - lat) / Osm2XConversions.METER_TO_LATLON;
049
050                return xy;
051        }
052}
