001package de.sgd.josm.plugins.osm2x.io.svg;
002
003import de.sgd.josm.plugins.osm2x.Osm2XConversions;
004
005public class SvgCircle extends SvgShape {
006
007        private SvgPoint center;
008        private double radius;
009
010        /**
011         *
012         * @param center
013         * @param radius in meters
014         */
015        public SvgCircle(SvgPoint center, double radius) {
016                this.center = center;
017                this.radius = radius;
018        }
019
020        /**
021         *
022         * @param center
023         * @param radius in meters
024         */
025        public SvgCircle(SvgPoint center, double radius, String fill) {
026                this(center, radius);
027                this.fill = fill;
028        }
029
030        /**
031         *
032         * @param center
033         * @param radius in meters
034         */
035        public SvgCircle(SvgPoint center, double radius, String stroke, double stroke_width) {
036                this(center, radius);
037        }
038
039        @Override
040        public String toSvg(SvgPoint documentOrigin) {
041                double[] c = center.getLocalCoordinates(documentOrigin);
042
043                String svg = String.format("<circle cx=\"%.3f\" cy=\"%.3f\" r=\"%.3f\" ", c[0], c[1], radius);
044
045                svg = svg.concat(String.format("fill=\"%s\" stroke=\"%s\" stroke-width=\"%.3f\"", fill, stroke, stroke_width));
046
047                return svg.concat("/>\n");
048        }
049
050        @Override
051        double minLat() {
052                return center.lat() - radius * Osm2XConversions.METER_TO_LATLON;
053        }
054
055        @Override
056        double maxLat() {
057                return center.lat() + radius * Osm2XConversions.METER_TO_LATLON;
058        }
059
060        @Override
061        double minLon() {
062                return center.lon() - radius * Osm2XConversions.METER_TO_LATLON / Math.cos(center.lat()*Math.PI/180);
063        }
064
065        @Override
066        double maxLon() {
067                return center.lon() + radius * Osm2XConversions.METER_TO_LATLON / Math.cos(center.lat()*Math.PI/180);
068        }
069
070}
