001package de.sgd.josm.plugins.osm2x.io;
002
003import static org.openstreetmap.josm.tools.I18n.tr;
004
005import java.io.File;
006import java.io.FileWriter;
007import java.io.IOException;
008
009import org.openstreetmap.josm.actions.ExtensionFileFilter;
010import org.openstreetmap.josm.gui.io.importexport.OsmExporter;
011import org.openstreetmap.josm.gui.layer.OsmDataLayer;
012
013public class NavExporter extends OsmExporter {
014
015        public NavExporter() {
016                super(new ExtensionFileFilter("nav", "nav", tr("SGD navigation files") + " (*.nav)"));
017        }
018
019        @Override
020        protected void doSave(File file, OsmDataLayer layer) throws IOException {
021                // Save map data to *.nav file and adresses to *.adr file
022                String adr_path = file.getAbsolutePath();
023                adr_path = adr_path.replaceFirst(".nav", ".adr");
024                File adr_file = new File(adr_path);
025
026                try (FileWriter writer = new FileWriter(file);
027                                NavWriter w = new NavWriter(writer);)
028                {
029                        layer.data.getReadLock().lock();
030                        try {
031                                w.writeDataset(layer.getDataSet());
032                        } finally {
033                                layer.data.getReadLock().unlock();
034                        }
035                }
036
037                try (FileWriter writer = new FileWriter(adr_file);
038                                AdrWriter w = new AdrWriter(writer);)
039                {
040                        layer.data.getReadLock().lock();
041                        try
042                        {
043                                w.writeDataset(layer.getDataSet());
044                        } finally {
045                                layer.data.getReadLock().unlock();
046                        }
047                }
048        }
049}
