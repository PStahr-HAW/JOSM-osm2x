001package de.sgd.josm.plugins.osm2x.io;
002
003import java.io.File;
004import java.io.FileWriter;
005import java.io.IOException;
006import java.util.ArrayList;
007
008import org.openstreetmap.josm.actions.ExtensionFileFilter;
009import org.openstreetmap.josm.data.osm.Node;
010import org.openstreetmap.josm.data.osm.Way;
011import org.openstreetmap.josm.gui.io.importexport.OsmExporter;
012import org.openstreetmap.josm.gui.layer.OsmDataLayer;
013
014import de.sgd.josm.plugins.osm2x.io.svg.SvgCircle;
015import de.sgd.josm.plugins.osm2x.io.svg.SvgDocument;
016import de.sgd.josm.plugins.osm2x.io.svg.SvgPath;
017import de.sgd.josm.plugins.osm2x.io.svg.SvgPoint;
018
019public class SvgExporter extends OsmExporter {
020
021        public SvgExporter() {
022                super(new ExtensionFileFilter("svg", "svg", "Scalable Vector Graphics (SGD) (*.svg)"));
023        }
024
025        @Override
026        protected void doSave(File file, OsmDataLayer layer) throws IOException {
027                // Save map data to *.svg file and create .yaml config file
028                String svg_path = file.getAbsolutePath();
029                if (!svg_path.endsWith(".svg"))
030                {
031                        svg_path = svg_path.concat(".svg");
032                }
033
034                // get data from map and convert to svg
035                SvgDocument doc = new SvgDocument();
036                layer.data.getReadLock().lock();
037                try {
038                        // get all nodes contained in dataset
039                        for (Node n : layer.getDataSet().getNodes())
040                        {
041                                // if node is part of a way skip this node. It will be added with the way
042                                if (!n.isDeleted() && n.hasKey("barrier", "natural"))
043                                {
044                                        SvgCircle c = new SvgCircle(new SvgPoint(n.lat(), n.lon()), 0.1, "black");
045                                        doc.addGeometry(c);
046                                }
047                        }
048
049                        // get all ways from dataset
050                        for (Way w : layer.getDataSet().getWays())
051                        {
052                                ArrayList<SvgPoint> waypoints = new ArrayList<>();
053                                for (Node n : w.getNodes()) {
054                                        waypoints.add(new SvgPoint(n.lat(), n.lon()));
055                                }
056                                if (waypoints.size() < 2) continue;     // a way must contain at least two points
057
058                                if (w.isClosed()) {
059                                        // create closed path with fill
060                                        doc.addGeometry(new SvgPath(waypoints, "black"));
061                                } else {
062                                        // create path with stroke
063                                        doc.addGeometry(new SvgPath(waypoints, "black", 0.2));
064                                }
065                        }
066                } finally {
067                        // unlock layer
068                        layer.data.getReadLock().unlock();
069                }
070
071                try (FileWriter svgWriter = new FileWriter(svg_path);
072                                SvgWriter w = new SvgWriter(svgWriter);)
073                {
074                        w.writeDataset(doc);
075                }
076
077                String yaml_path = svg_path.replaceFirst(".svg", ".yaml");
078                try (FileWriter writer = new FileWriter(yaml_path);
079                                YamlWriter w = new YamlWriter(writer);)
080                {
081                        w.writeDataset(doc);
082                }
083        }
084}
