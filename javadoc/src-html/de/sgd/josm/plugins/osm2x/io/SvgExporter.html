001package de.sgd.josm.plugins.osm2x.io;
002
003import java.io.File;
004import java.io.FileWriter;
005import java.io.IOException;
006
007import org.openstreetmap.josm.actions.ExtensionFileFilter;
008import org.openstreetmap.josm.gui.io.importexport.OsmExporter;
009import org.openstreetmap.josm.gui.layer.OsmDataLayer;
010
011public class SvgExporter extends OsmExporter {
012
013        public SvgExporter() {
014                super(new ExtensionFileFilter("svg", "svg", "Scalable Vector Graphics (SGD) (*.svg)"));
015        }
016
017        @Override
018        protected void doSave(File file, OsmDataLayer layer) throws IOException {
019                // Save map data to *.svg file and create .yaml config file
020                String adr_path = file.getAbsolutePath();
021                adr_path = adr_path.replaceFirst(".svg", ".yaml");
022                File adr_file = new File(adr_path);
023
024                try (FileWriter writer = new FileWriter(file);
025                                SvgWriter w = new SvgWriter(writer);)
026                {
027                        layer.data.getReadLock().lock();
028                        try {
029                                w.writeDataset(layer.getDataSet());
030                        } finally {
031                                layer.data.getReadLock().unlock();
032                        }
033                }
034
035                /*
036                try (FileWriter writer = new FileWriter(adr_file);
037                                AdrWriter w = new AdrWriter(writer);)
038                {
039                        layer.data.getReadLock().lock();
040                        try
041                        {
042                                w.writeDataset(layer.getDataSet());
043                        } finally {
044                                layer.data.getReadLock().unlock();
045                        }
046                }*/
047        }
048}
