001package de.sgd.josm.plugins.osm2x.io.svg;
002
003import java.io.IOException;
004import java.io.Writer;
005import java.util.ArrayList;
006
007public class SvgDocument {
008
009        ArrayList<SvgShape> shapes;
010
011        double minLon, minLat, maxLon, maxLat;
012
013        public SvgDocument() {
014                shapes = new ArrayList<>();
015
016                // set lon/lat to max values
017                minLon = 180;
018                maxLon = -180;
019                minLat = 90;
020                maxLat = -90;
021        }
022
023        public void addGeometry(SvgShape geometry) {
024                shapes.add(geometry);
025
026                if (geometry.minLon() < minLon) {
027                        minLon = geometry.minLon();
028                } else if (geometry.maxLon() > maxLon) {
029                        maxLon = geometry.maxLon();
030                }
031
032                if (geometry.minLat() < minLat) {
033                        minLat = geometry.minLat();
034                } else if (geometry.maxLat() > maxLat) {
035                        maxLat = geometry.maxLat();
036                }
037        }
038
039        public void writeToFile(Writer writer) throws IOException {
040                // Origin for svg/inkscape
041                SvgPoint documentOrigin = new SvgPoint(maxLat, minLon);
042                double[] xy = getDocSize();
043
044                // write head
045                writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
046                String line = String.format("<svg width=\"%dpx\" "
047                                + "height=\"%dpx\" "
048                                + "viewport=\"0 0 %1$d %2$d\" "
049                                + "version=\"1.1\" "
050                                + "xmlns=\"http://www.w3.org/2000/svg\" "
051                                + "xmlns:svg=\"http://www.w3.org/2000/svg\">\n",
052                                (int)Math.ceil(xy[0]), (int)Math.ceil(xy[1]));
053                writer.write(line);
054                writer.write(String.format("  <desc>Document origin (bottom-left corner): lat=%.7f, lon=%.7f</desc>\n\n",
055                                minLat, minLon));
056
057                // write body
058                String indent = "  ";
059                for (SvgShape shp : shapes) {
060                        writer.write(indent + shp.toSvg(documentOrigin));
061                }
062
063                writer.write("</svg>\n");
064        }
065
066        /**
067         * Get the document size in meters.
068         * @return array [width, height] in meters
069         */
070        public double[] getDocSize() {
071                SvgPoint documentOrigin = new SvgPoint(maxLat, minLon);
072                SvgPoint pnt = new SvgPoint(minLat, maxLon);
073                return pnt.getLocalCoordinates(documentOrigin);
074        }
075
076        /**
077         * Get the global coordinates of the bottom left corner of the map.
078         * @return the global coordinates of the bottom left corner of the map.
079         */
080        public double[] getMapOrigin() {
081                double[] d = {minLat, minLon};
082                return d;
083        }
084}
