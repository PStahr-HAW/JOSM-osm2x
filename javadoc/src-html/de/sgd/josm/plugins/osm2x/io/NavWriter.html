001package de.sgd.josm.plugins.osm2x.io;
002
003import java.io.Closeable;
004import java.io.FileWriter;
005import java.io.IOException;
006import java.util.List;
007
008import org.openstreetmap.josm.data.osm.DataSet;
009import org.openstreetmap.josm.data.osm.Node;
010import org.openstreetmap.josm.data.osm.Way;
011
012public class NavWriter implements Closeable {
013
014        FileWriter writer;
015
016        public NavWriter(FileWriter writer) {
017                this.writer = writer;
018        }
019
020        public void writeDataset(DataSet ds) throws IOException {
021                // TODO write dataset to file
022                if (ds != null)
023                {
024                        writer.write("<?xml version=\"1.0\" encoding=\"utf-8\"?>\n");
025                        writer.write(String.format("<nodelist version=\"%s\">\n", ds.getVersion()));
026
027                        for (Node n : ds.getNodes())
028                        {
029                                List<Way> parentWays = n.getParentWays();
030                                if (parentWays.size() > 1)
031                                {
032                                        writer.write(String.format("  <node id=\"%d\" lat=\"%.7f\" lon=\"%.7f\">\n",
033                                                        n.getId(), n.lat(), n.lon()));
034
035                                        for (Way w : n.getParentWays())
036                                        {
037                                                String highway = w.get("highway");
038                                                String surface = w.get("surface");
039
040                                                for (Node nn : w.getNeighbours(n))
041                                                {
042                                                        writer.write(String.format("    <nd ref=\"%d\">\n", nn.getId()));
043                                                        writer.write(String.format("      <highway>%s</highway>\n", highway));
044                                                        writer.write(String.format("      <surface>%s</surface>\n", surface));
045                                                        writer.write("    </nd>\n");
046                                                }
047                                        }
048
049                                        writer.write("  </node>\n");
050                                }
051                        }
052
053                        writer.write("</nodelist>\n");
054                }
055        }
056
057        @Override
058        public void close() throws IOException {
059                // TODO Auto-generated method stub
060
061        }
062
063}
