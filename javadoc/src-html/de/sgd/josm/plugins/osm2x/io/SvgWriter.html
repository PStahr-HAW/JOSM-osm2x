001package de.sgd.josm.plugins.osm2x.io;
002
003import java.io.Closeable;
004import java.io.FileWriter;
005import java.io.IOException;
006import java.util.ArrayList;
007
008import org.openstreetmap.josm.data.Bounds;
009import org.openstreetmap.josm.data.coor.LatLon;
010import org.openstreetmap.josm.data.osm.DataSet;
011import org.openstreetmap.josm.data.osm.Node;
012import org.openstreetmap.josm.data.osm.Way;
013import org.openstreetmap.josm.gui.MainApplication;
014
015import de.sgd.josm.plugins.osm2x.io.svg.SvgCircle;
016import de.sgd.josm.plugins.osm2x.io.svg.SvgDocument;
017import de.sgd.josm.plugins.osm2x.io.svg.SvgPath;
018import de.sgd.josm.plugins.osm2x.io.svg.SvgPoint;
019
020public class SvgWriter implements Closeable {
021
022        FileWriter writer;
023        LatLon origin;
024
025        public SvgWriter(FileWriter writer) {
026                this.writer = writer;
027        }
028
029        public void writeDataset(DataSet ds) throws IOException {
030                if (ds != null)
031                {
032                        Bounds pb = MainApplication.getMap().mapView.getRealBounds();
033
034                        System.out.printf("RealBounds: %.7f %.7f %.7f %.7f\n", pb.getMinLat(), pb.getMaxLat(), pb.getMinLon(), pb.getMaxLon());
035
036                        //this.origin = new LatLon(pb.maxNorth, pb.minEast);
037                        this.origin = new LatLon(pb.getMaxLat(), pb.getMinLon());
038                        System.out.println("Origin coordinates: " + origin.toDisplayString());
039
040                        SvgDocument doc = new SvgDocument();
041
042
043                        // create svg file from layer
044                        for (Node n : ds.getNodes())
045                        {
046                                // if node is part of a way skip this node
047                                if (!n.isDeleted() && n.hasKey("barrier", "natural"))
048                                {
049                                        SvgCircle c = new SvgCircle(new SvgPoint(n.lat(), n.lon()), 0.1, "black");
050                                        doc.addGeometry(c);
051                                }
052                        }
053
054                        for (Way w : ds.getWays())
055                        {
056                                ArrayList<SvgPoint> waypoints = new ArrayList<>();
057                                for (Node n : w.getNodes()) {
058                                        waypoints.add(new SvgPoint(n.lat(), n.lon()));
059                                }
060                                if (waypoints.size() < 2) continue;
061
062                                if (w.isClosed()) {
063                                        // create closed path with fill
064                                        doc.addGeometry(new SvgPath(waypoints, "black"));
065                                } else {
066                                        // create path with stroke
067                                        doc.addGeometry(new SvgPath(waypoints, "black", 0.2));
068                                }
069                        }
070
071                        doc.writeToFile(writer);
072                }
073        }
074
075        @Override
076        public void close() throws IOException {
077                // TODO Auto-generated method stub
078                writer.close();
079        }
080}
