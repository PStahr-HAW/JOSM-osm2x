043        /**
044         * Create a new Osm2XMesher object
045         * @param ds the dataset to edit
046         */
047        public Osm2XMesher(DataSet ds) {
048                this.originalDS = ds;
049                this.strToDouble = Pattern.compile("(\\d+\\.?\\d*)");
050                this.idGen = new IdGenerator(originalDS);
051
052                // Check if all paths have the width attribute
053                boolean width_present = true;
054                for (Way w : ds.getWays()) {
055                        if (!w.hasTag("width")) {
056                                // highlight paths without width attribute
057                                w.setHighlighted(true);
058                                width_present = false;
059                        }
060                }
061
062                if (!width_present) {
063                        this.ds = ds;
064                }
065        }

101        /**
102         * To allow changing the path in short intervals, ways with a length greater than a threshold are linearly interpolated.
103         */
104        public void interpolate() {
105                DataSet newDS = new DataSet(this.originalDS);
106
107                Logging.info("Start interpolation");
108
109                newDS.beginUpdate();
110                try
111                {
112                        for (Way w : newDS.getWays()) {
113                                for (int i = 1; i < w.getRealNodesCount(); i++) {
114                                        Node lastNode = w.getNode(i-1);
115                                        Node currNode = w.getNode(i);
116
117                                        double dist = lastNode.getCoor().greatCircleDistance(currNode.getCoor());
118
119                                        double interpDist = Config.getPref().getDouble(Osm2XConstants.PREF_INTERP_DIST, Osm2XConstants.DEFAULT_INTERP_DIST);
120                                        if (dist > interpDist)
121                                        {
122                                                int addNodes = (int) Math.ceil(dist / interpDist);
123
124                                                int k = 0;
125                                                for (k = 1; k < addNodes; k++) {
126                                                        // Calculate new latlon and node
127                                                        LatLon ll = lastNode.getCoor().interpolate(currNode.getCoor(), (double)k/addNodes);
128                                                        Node nn = new Node(ll);
129                                                        nn.setOsmId(this.idGen.generateID(), 1);
130                                                        // add new node to dataset and way
131                                                        newDS.addPrimitive(nn);
132                                                        w.addNode(i+k-1, nn);
133                                                }
134                                                i += k-1;
135                                        }
136                                }
137                        }
138                } finally
139                {
140                        newDS.endUpdate();
141                }
142                this.interpolatedDS = newDS;
143        }

145        /**
146         * Create parallel ways and connect them to original ways
147         */
148        public void createRoadNetwork() {
149                // get all ways -> ignore areas, but with warning
150                // ignore ways without width attribute
151                // ignore ways with attribute mesh=false
152                DataSet ds = (this.interpolatedDS == null) ? new DataSet(originalDS) : new DataSet(interpolatedDS);
153
154                System.out.println("Start meshing");
155
156                ds.beginUpdate();
157                try
158                {
159                        HashMap<Node, Osm2XNodeList> addedNodes = new HashMap<>();
160                        List<Way> addedWays = new ArrayList<>();
161
162                        for (Node n : ds.getNodes()) {
163                                Osm2XNodeList nodes = new Osm2XNodeList();
164                                // check that nodes have an id
165                                if (n.getId() == 0)
166                                {
167                                        n.setOsmId(this.idGen.generateID(), 1);
168                                }
169
170                                // set original node as parent id -> is copying all keys correct?
171                                Map<String, String> tags = n.getKeys();
172                                tags.put("pid", Long.toString(n.getId()));
173
174                                if (n.isConnectionNode()) {
175                                        // node connects two or more ways
176                                        TreeMap<Double, Way> angles = getNeighbourNodes(n);
177                                        Entry<Double, Way> last_entry = null;
178                                        for (Entry<Double, Way> entry : angles.entrySet()) {
179                                                if (last_entry == null) {
180                                                        last_entry = entry;
181                                                        continue;
182                                                }
183
184                                                Node new_node = new Node(calculateNode(n.getCoor(), last_entry, entry));
185                                                new_node.setOsmId(this.idGen.generateID(), 1);
186                                                new_node.setKeys(tags);
187                                                nodes.add(new_node, n.getCoor().bearing(new_node.getCoor()));
188                                                last_entry = entry;
189                                        }
190
191                                        Node new_node = new Node(calculateNode(n.getCoor(), last_entry, angles.firstEntry()));
192                                        new_node.setOsmId(this.idGen.generateID(), 1);
193                                        new_node.setKeys(tags);
194                                        nodes.add(new_node, n.getCoor().bearing(new_node.getCoor()));
195                                        addedNodes.put(n, nodes);
196                                }
197                        }
198
199                        // compute parallel ways
200                        List<Node> more_nodes = new ArrayList<>();
201                        List<Way> more_ways = new ArrayList<>();
202                        for (Way w : ds.getWays()) {
203                                if (w.getRealNodesCount() < 1) continue;
204
205                                double width = 0;
206                                if (w.get("width") != null) {
207                                        width = parseDouble(w.get("width"));
208                                        if (width < 1.5) continue;
209                                } else {
210                                        continue;
211                                }
212                                // TODO filtern??
213                                Way way_l = new Way(0);
214                                way_l.setKeys(w.getKeys());
215                                Map<String, String> keys_l = way_l.getKeys();
216                                keys_l.put("left", "true");
217
218                                Way con_l1 = new Way(0);
219                                con_l1.setKeys(keys_l);
220                                con_l1.addNode(w.getNode(0));
221
222                                Way con_l2 = new Way(0);
223                                con_l2.setKeys(keys_l);
224
225                                // add angle tag to left way
226                                keys_l.put("angle", "0.0");
227                                way_l.setKeys(keys_l);
228
229                                // ways for rights side
230                                Way way_r = new Way(0);
231
232                                Way con_r1 = new Way(0);
233                                con_r1.setKeys(w.getKeys());
234                                con_r1.addNode(w.getNode(0));
235
236                                Way con_r2 = new Way(0);
237                                con_r2.setKeys(w.getKeys());
238
239                                // add angle tag to right ways
240                                Map<String, String> keys_r = way_l.getKeys();
241                                keys_r.put("angle", "0.0");
242                                way_r.setKeys(keys_r);
243
244                                // add angle to middle ways
245                                Map<String, String> keys = w.getKeys();
246                                keys.put("angle", "0.0");
247                                w.setKeys(keys);
248
249                                Way[] con_r = {con_r1, con_r2};
250                                Way[] con_l = {con_l1, con_l2};
251
252                                if (addedNodes.containsKey(w.getNode(0)) && w.getNodesCount() > 1) {
253                                        double ang1 = w.getNode(0).getCoor().bearing(w.getNode(1).getCoor());
254                                        double ang2 = ang1 > Math.PI ? ang1 - Math.PI : ang1 + Math.PI;
255
256                                        List<Node> r_nodes = addedNodes.get(w.getNode(0)).getRightNodesWithinAngle(ang1, ang2);
257                                        for (Node n : r_nodes) {
258                                                way_r.addNode(n);
259                                        }
260                                        if (r_nodes.size() > 0) {
261                                                con_r[1].addNode(r_nodes.get(r_nodes.size()-1));
262                                        }
263
264                                        List<Node> l_nodes = addedNodes.get(w.getNode(0)).getLeftNodesWithinAngle(ang1, ang2);
265                                        for (Node n : l_nodes) {
266                                                way_l.addNode(n);
267                                        }
268                                        if (l_nodes.size() > 0) {
269                                                con_l[1].addNode(l_nodes.get(l_nodes.size()-1));
270                                        }
271                                }
272
273                                for (int i = 1; i < w.getNodesCount()-1; i++) {
274                                        Node n_i = w.getNode(i);
275                                        double ang1 = n_i.getCoor().bearing(w.getNode(i+1).getCoor());
276                                        double ang2 = n_i.getCoor().bearing(w.getNode(i-1).getCoor());
277
278                                        if (addedNodes.containsKey(n_i)) {
279                                                // add nodes to way
280                                                for (Node n : addedNodes.get(n_i).getRightNodesWithinAngle(ang1, ang2)) {
281                                                        way_r.addNode(n);
282                                                        con_r[(i+1) % 2].addNode(n);
283                                                }
284
285                                                for (Node n : addedNodes.get(n_i).getLeftNodesWithinAngle(ang1, ang2)) {
286                                                        way_l.addNode(n);
287                                                        con_l[(i+1) % 2].addNode(n);
288                                                }
289
290                                                con_r[i % 2].addNode(n_i);
291                                                con_l[i % 2].addNode(n_i);
292                                        } else {
293                                                double ang = (ang1 > ang2) ? (ang1 + ang2) / 2 + Math.PI : (ang1 + ang2) / 2;
294
295                                                // Way on the right side
296                                                Node new_node = new Node(Osm2XConversions.calculatePositionFrom(n_i.getCoor(), ang, width/3));
297                                                new_node.setOsmId(this.idGen.generateID(), 1);
298                                                Map<String, String> tags_ = n_i.getKeys();
299                                                tags_.put("pid", Long.toString(n_i.getId()));
300                                                new_node.setKeys(tags_);
301                                                more_nodes.add(new_node);
302                                                way_r.addNode(new_node);
303
304                                                // Way on the left side
305                                                Node new_node1 = new Node(Osm2XConversions.calculatePositionFrom(n_i.getCoor(), ang + Math.PI, width/3));
306                                                new_node1.setOsmId(this.idGen.generateID(), 1);
307                                                new_node1.setKeys(tags_);
308                                                more_nodes.add(new_node1);
309                                                way_l.addNode(new_node1);
310
311                                                con_r[(i+1) % 2].addNode(new_node);
312                                                con_r[i % 2].addNode(n_i);
313                                                con_l[(i+1) % 2].addNode(new_node1);
314                                                con_l[i % 2].addNode(n_i);
315                                        }
316                                }
317
318                                Node last_node = w.getNode(w.getNodesCount()-1);
319                                con_r[(w.getNodesCount()-1) % 2].addNode(last_node);
320                                con_l[(w.getNodesCount()-1) % 2].addNode(last_node);
321
322                                if (addedNodes.containsKey(last_node)) {
323                                        double ang2 = last_node.getCoor().bearing(w.getNode(w.getNodesCount()-2).getCoor());
324                                        double ang1 = ang2 > Math.PI ? ang2 - Math.PI : ang2 + Math.PI;
325
326                                        List<Node> r_nodes = addedNodes.get(last_node).getRightNodesWithinAngle(ang1, ang2);
327                                        for (Node n : r_nodes) {
328                                                way_r.addNode(n);
329                                        }
330                                        if (r_nodes.size() > 0) {
331                                                con_r[w.getNodesCount() % 2].addNode(r_nodes.get(0));
332                                        }
333
334                                        List<Node> l_nodes = addedNodes.get(last_node).getLeftNodesWithinAngle(ang1, ang2);
335                                        for (Node n : l_nodes) {
336                                                way_l.addNode(n);
337                                        }
338                                        if (l_nodes.size() > 0) {
339                                                con_l[w.getNodesCount() % 2].addNode(l_nodes.get(0));
340                                        }
341                                }
342
343                                more_ways.add(way_l);
344                                more_ways.add(con_l[0]);
345                                more_ways.add(con_l[1]);
346                                more_ways.add(way_r);
347                                more_ways.add(con_r[0]);
348                                more_ways.add(con_r[1]);
349                        }
350
351                        System.out.println("Add all nodes to dataset");
352                        for (Osm2XNodeList nodes : addedNodes.values()) {
353                                for (Node n : nodes.getNodelist()) {
354                                        ds.addPrimitive(n);
355                                }
356                        }
357                        System.out.println("Add all ways to dataset");
358                        for (Way w : addedWays) {
359                                ds.addPrimitive(w);
360                        }
361
362                        System.out.println("Add more nodes to dataset");
363                        for (Node n : more_nodes) {
364                                ds.addPrimitive(n);
365                        }
366                        System.out.println("Add more ways to dataset");
367                        for (Way w : more_ways) {
368                                ds.addPrimitive(w);
369                        }
370
371                } catch (Exception e) {
372                        System.out.println(e.getMessage());
373                } finally
374                {
375                        ds.endUpdate();
376                }
377                this.ds = ds;
378        }

388        /**
389         * Get all neighbour nodes to node with an angle
390         * @param node
391         * @return
392         */
393        private TreeMap<Double, Way> getNeighbourNodes(Node node) {
394                TreeMap<Double, Way> neighbours = new TreeMap<>();
395
396                for (Way w : node.getParentWays()) {
397                        for (Node n : w.getNeighbours(node)) {
398                                neighbours.put(node.getCoor().bearing(n.getCoor()), w);
399                        }
400                }
401                return neighbours;
402        }

404        /**
405         * Calculate a new node based on the width of way 1 and way 2.
406         * @param origin the node connecting both ways
407         * @param way1 the first way
408         * @param way2 the second way
409         * @return coordinates for the calculated node
410         */
411        private LatLon calculateNode(LatLon origin, Entry<Double, Way> way1, Entry<Double, Way> way2) {
412                // calculate distance in meters
413                double b1 = parseDouble(way1.getValue().get("width"));
414                double b2 = parseDouble(way2.getValue().get("width"));
415                double deltaAng = way2.getKey() - way1.getKey();
416
417                double x = (b1 * Math.sin(way2.getKey()) + b2 * Math.sin(way1.getKey()))
418                                / 3.0 / Math.sin(deltaAng);
419
420                double y = (b1 * Math.cos(way2.getKey()) + b2 * Math.cos(way1.getKey()))
421                                / 3.0 / Math.sin(deltaAng);
422
423                x = origin.lon() + x*Osm2XConversions.METER_TO_LATLON/Math.cos(origin.lat()/180*Math.PI);
424                y = origin.lat() + y*Osm2XConversions.METER_TO_LATLON;
425
426                return new LatLon(y,x);
427        }

429        /**
430         * Convert string to double without throwing an exception
431         * @param s the string to parse
432         * @return the number
433         */
434        private double parseDouble(String s) {
435                double d = 0;
436                if (s == null) return d;
437                try {
438                        d = Double.valueOf(s);
439                } catch (NumberFormatException nfe) {
440                        // try to parse with regex
441                        Matcher matcher = strToDouble.matcher(s);
442                        if (matcher.find()) {
443                                try {
444                                        d = Double.valueOf(matcher.group());
445                                } catch (NumberFormatException nfe1) {
446                                        Logging.error(nfe1);
447                                }
448                        }
449                }
450                return d;
451        }
