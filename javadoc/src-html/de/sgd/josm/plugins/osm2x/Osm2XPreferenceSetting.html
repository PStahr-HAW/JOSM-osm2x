001package de.sgd.josm.plugins.osm2x;
002
003import static org.openstreetmap.josm.tools.I18n.tr;
004
005import java.awt.GridBagLayout;
006
007import javax.swing.BorderFactory;
008import javax.swing.Box;
009import javax.swing.JLabel;
010import javax.swing.JPanel;
011import javax.swing.JScrollPane;
012import javax.swing.JSpinner;
013import javax.swing.SpinnerNumberModel;
014
015import org.openstreetmap.josm.gui.preferences.DefaultTabPreferenceSetting;
016import org.openstreetmap.josm.gui.preferences.PreferenceTabbedPane;
017import org.openstreetmap.josm.spi.preferences.Config;
018import org.openstreetmap.josm.tools.GBC;
019
020import de.sgd.josm.plugins.osm2x.helper.Osm2XConstants;
021
022public class Osm2XPreferenceSetting extends DefaultTabPreferenceSetting {
023
024        public JPanel prefPanel;
025
026        private final JSpinner spinner = new JSpinner(new SpinnerNumberModel(5.0, 2.0, null, 0.1));
027
028        public Osm2XPreferenceSetting() {
029                super(Osm2XConstants.ICON_PREF_48,
030                                tr("Osm2X Preferences"),
031                                tr("Convert and export OSM data to be used on the Shared Guide Dog 4.0"));
032        }
033
034        protected JPanel createSettings() {
035                JPanel general = new JPanel(new GridBagLayout());
036                general.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 0));
037                general.setName(tr("Osm2X preferences"));
038
039                // options for interpolation
040                JLabel jLabelSpinner = new JLabel(tr("Maximum way length:"));
041                spinner.getModel().setValue(Config.getPref().getDouble(Osm2XConstants.PREF_INTERP_DIST, 5.0));
042                general.add(jLabelSpinner, GBC.std().insets(0, 5, 10, 0));
043                general.add(spinner, GBC.eol().fill(GBC.HORIZONTAL).insets(5, 5, 200, 5));
044
045                // end of dialog, scroll bar
046                general.add(Box.createVerticalGlue(), GBC.eol().fill(GBC.VERTICAL));
047
048                return general;
049        }
050
051        @Override
052        public void addGui(PreferenceTabbedPane gui) {
053                prefPanel = gui.createPreferenceTab(this);
054
055                JScrollPane scrollpane = new JScrollPane(createSettings());
056                scrollpane.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 0));
057                prefPanel.add(scrollpane, GBC.eol().fill(GBC.BOTH));
058        }
059
060        @Override
061        public boolean ok() {
062                Config.getPref().putDouble(Osm2XConstants.PREF_INTERP_DIST, (Double)spinner.getModel().getValue());
063                return false;
064        }
065
066}
