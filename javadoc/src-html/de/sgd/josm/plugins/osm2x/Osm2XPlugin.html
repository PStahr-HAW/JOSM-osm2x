001package de.sgd.josm.plugins.osm2x;
002
003import org.openstreetmap.josm.actions.ExtensionFileFilter;
004import org.openstreetmap.josm.gui.MapFrame;
005import org.openstreetmap.josm.gui.preferences.PreferenceSetting;
006import org.openstreetmap.josm.plugins.Plugin;
007import org.openstreetmap.josm.plugins.PluginInformation;
008
009import de.sgd.josm.plugins.osm2x.io.NavExporter;
010import de.sgd.josm.plugins.osm2x.io.SvgExporter;
011
012public class Osm2XPlugin extends Plugin {
013
014        //private IconToggleButton btn;
015        protected static Osm2XDialog osm2xDialog;
016
017        public Osm2XPlugin(PluginInformation info) {
018                super(info);
019                // init your plugin
020                ExtensionFileFilter.addExporter(new NavExporter());
021                ExtensionFileFilter.addExporter(new SvgExporter());
022        }
023
024        @Override
025        public void mapFrameInitialized(MapFrame oldFrame, MapFrame newFrame) {
026                // Plugins implementing a method mapFrameInitialized are notified about the change of the current map frame
027                if (newFrame != null) {
028                        newFrame.addToggleDialog(osm2xDialog = new Osm2XDialog());
029                } else {
030                        osm2xDialog = null;
031                }
032        }
033
034        @Override
035        public PreferenceSetting getPreferenceSetting() {
036                return new Osm2XPreferenceSetting();
037        }
038

040        //
041        //}
042}
